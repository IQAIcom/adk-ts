## 1. Using LangChain Tools

LangChain provides a rich ecosystem of tools for AI agents. Here's how to integrate them with ADK-TS:

<Steps>
  <Step>
    ### Install Dependencies

    ```bash
    npm install langchain @langchain/core
    ```

    For specific tools, install the required packages (e.g., `@langchain/community`).

  </Step>
  <Step>
    ### Import and Wrap

    ```typescript
    import { FunctionTool } from '@iqai/adk';
    import { SerpAPI } from '@langchain/community/tools';

    // Create a LangChain SerpAPI tool instance
    const serpApi = new SerpAPI(process.env.SERPAPI_API_KEY);

    // Wrap the LangChain tool as an ADK FunctionTool
    const serpTool = new FunctionTool(
      async (query) => {
        try {
          // Call the LangChain tool's _call method
          const result = await serpApi._call(query);
          return { success: true, result };
        } catch (error) {
          return {
            success: false,
            error: error instanceof Error ? error.message : String(error)
          };
        }
      },
      {
        name: 'serp_search',
        description: 'Search the web using SerpAPI'
      }
    );
    ```

  </Step>
  <Step>
    ### Add to Agent

    ```typescript
    import { createAgent } from '@iqai/adk';

    const agent = createAgent({
      name: 'LangChainAgent',
      model: 'gemini-pro',
      tools: [serpTool],
      systemPrompt: 'You are a helpful assistant with search capabilities.'
    });
    ```

  </Step>
</Steps>

### Example: Using LangChain's Tavily Search Tool

```typescript
import { FunctionTool } from "@iqai/adk";
import { TavilySearchResults } from "@langchain/community/tools/tavily_search";

// Create the LangChain tool
const tavilySearch = new TavilySearchResults({
  apiKey: process.env.TAVILY_API_KEY,
  maxResults: 5,
});

// Wrap for ADK
const tavilyTool = new FunctionTool(
  async (query) => {
    try {
      const results = await tavilySearch._call(query);
      return { results };
    } catch (error) {
      return { error: `Search failed: ${error.message}` };
    }
  },
  {
    name: "tavily_search",
    description: "Search the web for current information using Tavily",
  }
);
```

## 2. Using CrewAI Tools

CrewAI provides specialized tools for AI agent capabilities. Here's how to integrate them with ADK-TS:

<Steps>
  <Step>
    ### Install Dependencies

    ```bash
    npm install crewai-tools
    ```

  </Step>
  <Step>
    ### Import and Create Tools

    ```typescript
    import { BaseTool } from '@iqai/adk';
    import { SerperDev } from 'crewai-tools';

    class SerperSearchTool extends BaseTool {
      private serperTool: SerperDev;

      constructor(apiKey: string) {
        super({
          name: 'serper_search',
          description: 'Search Google using Serper API for real-time results'
        });
        this.serperTool = new SerperDev({
          apiKey,
          searchType: 'search',
          numResults: 10
        });
      }

      async runAsync(args: { query: string; type?: string }) {
        try {
          const results = await this.serperTool.search(args.query, args.type || 'search');
          return { success: true, results };
        } catch (error) {
          return {
            success: false,
            error: `Search failed: ${error instanceof Error ? error.message : String(error)}`
          };
        }
      }

      getDeclaration() {
        return {
          name: this.name,
          description: this.description,
          parameters: {
            type: 'object',
            properties: {
              query: {
                type: 'string',
                description: 'Search query'
              },
              type: {
                type: 'string',
                description: 'Type of search (search, images, news)',
                enum: ['search', 'images', 'news']
              }
            },
            required: ['query']
          }
        };
      }
    }
    ```

  </Step>
  <Step>
    ### Add to Agent

    ```typescript
    import { createAgent } from '@iqai/adk';

    const serperTool = new SerperSearchTool(process.env.SERPER_API_KEY!);

    const agent = createAgent({
      name: 'NewsSearchAgent',
      model: 'gemini-pro',
      tools: [serperTool],
      systemPrompt: 'You are a helpful assistant with internet search capabilities.'
    });
    ```

  </Step>
</Steps>

## 3. Using Model Context Protocol (MCP)

ADK-TS provides integration with the Model Context Protocol, enabling compatibility with any MCP-compatible service:

```typescript
import { McpToolset } from "@iqai/adk";
import { Client } from "@modelcontextprotocol/sdk/client";

// Initialize the MCP client
const client = new Client({
  baseUrl: "https://your-mcp-server.com",
  apiKey: process.env.MCP_API_KEY,
});

// Create the toolset
const mcpToolset = new McpToolset(client);

// Get available tools
const availableTools = await mcpToolset.getAvailableTools();

// Convert tools for use in ADK
const selectedTools = await mcpToolset.convertToolDefinitions([
  "data_analysis",
  "document_retrieval",
]);

// Use in agent
import { createAgent } from "@iqai/adk";

const agent = createAgent({
  name: "McpAgent",
  tools: selectedTools,
  model: "gemini-pro",
});
```

## 4. OpenAI Integration Example

Integrating OpenAI's tools is simple with ADK-TS:

```typescript
import { BaseTool } from "@iqai/adk";
import OpenAI from "openai";

class OpenAIImageTool extends BaseTool {
  private client: OpenAI;

  constructor(apiKey: string) {
    super({
      name: "generate_image",
      description: "Generate images using OpenAI DALL-E",
    });
    this.client = new OpenAI({ apiKey });
  }

  async runAsync(args: { prompt: string; size?: string }) {
    try {
      const response = await this.client.images.generate({
        model: "dall-e-3",
        prompt: args.prompt,
        size: args.size || "1024x1024",
        n: 1,
      });

      return {
        url: response.data[0].url,
        revisedPrompt: response.data[0].revised_prompt,
      };
    } catch (error) {
      return {
        error: `Image generation failed: ${
          error instanceof Error ? error.message : String(error)
        }`,
      };
    }
  }

  getDeclaration() {
    return {
      name: this.name,
      description: this.description,
      parameters: {
        type: "object",
        properties: {
          prompt: {
            type: "string",
            description: "Detailed description of the image to generate",
          },
          size: {
            type: "string",
            enum: ["256x256", "512x512", "1024x1024"],
            description: "Size of the generated image",
          },
        },
        required: ["prompt"],
      },
    };
  }
}
```

## Custom Tool Class Pattern

For more sophisticated integrations, create custom tool classes that extend BaseTool.

### Basic Custom Tool

```typescript
import { BaseTool } from "@iqai/adk";
import { ExternalService } from "external-library";

interface ExternalToolConfig {
  apiKey: string;
  baseUrl?: string;
  timeout?: number;
}

class ExternalTool extends BaseTool {
  private client: ExternalService;

  constructor(config: ExternalToolConfig) {
    super({
      name: "external_tool",
      description: "Integration with external service",
    });

    this.client = new ExternalService({
      apiKey: config.apiKey,
      baseUrl: config.baseUrl || "https://api.example.com",
      timeout: config.timeout || 30000,
    });
  }

  async runAsync(args: { operation: string; parameters: Record<string, any> }) {
    try {
      switch (args.operation) {
        case "search":
          return await this.client.search(args.parameters);
        case "create":
          return await this.client.create(args.parameters);
        case "update":
          return await this.client.update(args.parameters);
        default:
          return { error: `Unknown operation: ${args.operation}` };
      }
    } catch (error) {
      return {
        error: `External service error: ${
          error instanceof Error ? error.message : String(error)
        }`,
      };
    }
  }

  getDeclaration() {
    return {
      name: this.name,
      description: this.description,
      parameters: {
        type: "object",
        properties: {
          operation: {
            type: "string",
            description: "Operation to perform",
            enum: ["search", "create", "update"],
          },
          parameters: {
            type: "object",
            description: "Parameters for the operation",
          },
        },
        required: ["operation", "parameters"],
      },
    };
  }
}
```
